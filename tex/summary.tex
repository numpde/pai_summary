\subsubsection*{MDP} 

$
	Q^V := 
	r + \gamma \sum_{x'} p(x' \given \slot) V(x')
$
\quad
$
	\pi^Q := \arg\max_a Q(\slot, a)
$
\quad
$
	V^\pi \stackrel{!}{=} Q^{V^\pi}(\slot, \pi(\slot))
$
\quad
\hfill
Bellman:
$
	V^\star
	=
	\max_a
	Q^{V^\star}(\slot, a)
$

\subsubsection*{DQN}

$
	\min_\theta
	\sum_{xarx'}
	[
		(r + \gamma \max_{a'} Q(x', a'; \theta^\text{old}))
		-
		Q(x, a; \theta)
	]^2
$
